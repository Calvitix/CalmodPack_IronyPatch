can_colonize_with_pop = {
    has_colonization_control = {
        value = no
        country = root
    }
    OR = {
        NOT = {
            has_trait = trait_self_modified
        }
        is_same_species = root
    }
    if = {
        limit = {
            OR = {
                is_sapient = no
            }
        }
        custom_tooltip = {
            always = no
        }
    }
    else = {
        always = yes
    }
    NOT = {
        has_trait = trait_plentiful_traditions_cocoon
    }
}
