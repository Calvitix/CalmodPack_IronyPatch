strike_craft_tech_check = {
    if = {
        limit = {
            has_technology = "nhsc_tech_strikecrafts_mercenary_2"
            NOT = {
                has_technology = "nhsc_tech_strikecrafts_mercenary_1"
            }
        }
        give_technology = {
            message = no
            tech = nhsc_tech_strikecrafts_mercenary_1
        }
    }
    if = {
        limit = {
            has_technology = "nhsc_tech_siegebreaker_strikecrafts"
            NOT = {
                has_technology = "nhsc_tech_tactical_strikecrafts"
            }
        }
        give_technology = {
            message = no
            tech = nhsc_tech_tactical_strikecrafts
        }
    }
    if = {
        limit = {
            OR = {
                has_technology = "nhsc_tech_tactical_strikecrafts"
                has_technology = "nhsc_tech_nanite_strikecraft"
                has_technology = "nhsc_tech_psionic_strikecraft"
                has_technology = "nhsc_tech_gravitic_strikecrafts"
                has_technology = "nhsc_tech_strikecrafts_mercenary_2"
            }
            NOT = {
                has_technology = "tech_strike_craft_3"
            }
        }
        give_technology = {
            message = no
            tech = tech_strike_craft_3
        }
    }
    if = {
        limit = {
            OR = {
                has_technology = "tech_strike_craft_3"
                has_technology = "nhsc_tech_strikecrafts_mercenary_1"
            }
            NOT = {
                has_technology = "tech_strike_craft_2"
            }
        }
        give_technology = {
            message = no
            tech = tech_strike_craft_2
        }
    }
    if = {
        limit = {
            has_technology = "tech_strike_craft_2"
            NOT = {
                has_technology = "tech_strike_craft_1"
            }
        }
        give_technology = {
            message = no
            tech = tech_strike_craft_1
        }
    }
    if = {
        limit = {
            has_technology = "nhsc_tech_nanite_strikecraft"
            NOT = {
                has_technology = "nhsc_tech_synthetic_lifeform_studies"
            }
        }
        give_technology = {
            message = no
            tech = nhsc_tech_synthetic_lifeform_studies
        }
    }
    if = {
        limit = {
            has_technology = "nhsc_tech_gravitic_strikecrafts"
            NOT = {
                has_technology = "nhsc_tech_gravity_weapons"
            }
        }
        give_technology = {
            message = no
            tech = nhsc_tech_gravity_weapons
        }
    }
    if = {
        limit = {
            has_technology = "nhsc_tech_psionic_strikecraft"
            NOT = {
                has_authority = auth_machine_intelligence
            }
            NOT = {
                has_technology = "nhsc_tech_psionic_weapons"
            }
        }
        give_technology = {
            message = no
            tech = nhsc_tech_psionic_weapons
        }
    }
}
