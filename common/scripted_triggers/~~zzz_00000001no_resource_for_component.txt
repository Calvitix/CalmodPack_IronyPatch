no_resource_for_component = {
    is_country_type = default
    OR = {
        NOT = {
            can_buy_on_market = $RESOURCE$
        }
        AND = {
            is_ai = yes
            has_monthly_income = {
                resource = $RESOURCE$
                value < -2
            }
            resource_stockpile_compare = {
                resource = $RESOURCE$
                value < 150
            }
        }
    }
}
